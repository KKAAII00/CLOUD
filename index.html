import java.util.Scanner; public class PenangMuseum { public static void main(String[] args) { Scanner input = new Scanner(System.in); int choice; int totalVisitors = 0; do { System.out.println("\n======================================"); System.out.println(" PENANG HERITAGE MUSEUM"); System.out.println("======================================"); System.out.println("1. Home"); System.out.println("2. History"); System.out.println("3. Gallery"); System.out.println("4. Ticket Booking"); System.out.println("5. Visitor Statistics"); System.out.println("6. About"); System.out.println("7. Exit"); System.out.println("======================================"); System.out.print("Enter your choice: "); choice = input.nextInt(); switch (choice) { case 1: System.out.println("\n========== HOME =========="); System.out.println("Welcome to Penang Heritage Museum!"); System.out.println("Explore the history and culture of Penang."); break; case 2: System.out.println("\n========== HISTORY =========="); System.out.println("1786 - Captain Francis Light established Penang."); System.out.println("1826 - Penang joined the Straits Settlements."); System.out.println("1941 - Japanese Occupation during World War II."); System.out.println("1957 - Independence of Malaya."); System.out.println("2008 - George Town became a UNESCO World Heritage Site."); break; case 3: System.out.println("\n========== GALLERY =========="); System.out.println("1.
