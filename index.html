import javax.swing.*; import java.awt.*; public class PenangMuseum { public static void main(String[] args) { JFrame frame = new JFrame("Penang Heritage Museum"); frame.setSize(800, 600); frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE); JPanel panel = new JPanel(); panel.setLayout(new BorderLayout()); JLabel title = new JLabel("Penang Heritage Museum", JLabel.CENTER); title.setFont(new Font("Arial", Font.BOLD, 28)); JTextArea info = new JTextArea(); info.setEditable(false); info.setLineWrap(true); info.setWrapStyleWord(true); info.setText( "Welcome to the Penang Heritage Museum\n\n" + "History of Penang:\n" + "• 1786 - Captain Francis Light founded Penang.\n" + "• 1826 - Became part of the Straits Settlements.\n" + "• 1941 - Japanese occupation during World War II.\n" + "• 2008 - George Town became a UNESCO World Heritage Site.\n\n" + "The museum preserves the rich cultural and historical heritage of Penang." ); JButton historyButton = new JButton("View History"); historyButton.addActionListener(e -> JOptionPane.showMessageDialog( frame, "Penang is one of Malaysia's most important historical trading ports." ) ); panel.add(title, BorderLayout.NORTH); panel.add(new JScrollPane(info), BorderLayout.CENTER); panel.add(historyButton, BorderLayout.SOUTH); frame.add(panel); frame.setVisible(true); } }
